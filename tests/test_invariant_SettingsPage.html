#include #include #include #include #include #include #include #include #include #define SERVER_HOST "127.0.0.1" #define SERVER_PORT 80 #define CONNECT_TIMEOUT_SEC 2 #define RECV_TIMEOUT_SEC 5 #define BUFFER_SIZE 4096 /* Attempt to connect to the server; returns fd or -1 if unavailable */ static int try_connect(void) { int fd = socket(AF_INET, SOCK_STREAM, 0); if (fd < 0) return -1; struct timeval tv; tv.tv_sec = CONNECT_TIMEOUT_SEC; tv.tv_usec = 0; setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv)); setsockopt(fd, SOL_SOCKET, SO_SNDTIMEO, &tv, sizeof(tv)); struct sockaddr_in addr; memset(&addr, 0, sizeof(addr)); addr.sin_family = AF_INET; addr.sin_port = htons(SERVER_PORT); addr.sin_addr.s_addr = inet_addr(SERVER_HOST); if (connect(fd, (struct sockaddr *)&addr, sizeof(addr)) < 0) { close(fd); return -1; } return fd; } /* * Send an HTTP request and return the HTTP status code, or -1 on error. * If the server is not reachable the test is skipped (returns -2). */ static int send_http_request(const char *request) { int fd = try_connect(); if (fd < 0) return -2; /* server not available – skip */ ssize_t sent = send(fd, request, strlen(request), 0); if (sent < 0) { close(fd); return -1; } char buf[BUFFER_SIZE]; memset(buf, 0, sizeof(buf)); ssize_t received = recv(fd, buf, sizeof(buf) - 1, 0); close(fd); if (received <= 0) return -1; /* Parse "HTTP/1.x NNN " */ int status = 0; if (sscanf(buf, "HTTP/%*d.%*d %d", &status) != 1) return -1; return status; } /* Build a raw HTTP GET request with an optional Authorization header */ static void build_request(char *out, size_t out_size, const char *path, const char *auth_header /* NULL = omit */) { if (auth_header && strlen(auth_header) > 0) { snprintf(out, out_size, "GET %s HTTP/1.1\r\n" "Host: %s\r\n" "%s\r\n" "Connection: close\r\n" "\r\n", path, SERVER_HOST, auth_header); } else { snprintf(out, out_size, "GET %s HTTP/1.1\r\n" "Host: %s\r\n" "Connection: close\r\n" "\r\n", path, SERVER_HOST); } } /* ------------------------------------------------------------------ */ /* Test: unauthenticated requests to protected endpoints must be */ /* rejected with 401 or 403. */ /* ------------------------------------------------------------------ */ START_TEST(test_unauthenticated_requests_rejected) { /* * Invariant: Every request to a protected API endpoint that lacks * valid authentication credentials MUST receive HTTP 401 or 403. * A 200 response means the endpoint is unprotected (CWE-287). * * NOTE: Only GET-capable endpoints are tested here. POST-only routes * (e.g. /api/wifi/delete) are covered by * test_write_endpoints_reject_unauthenticated_post to avoid false * confidence from a 405 Method Not Allowed masking an auth bypass. */ /* Pairs of { endpoint, Authorization header (empty string = omit) } */ struct { const char *path; const char *auth_header; const char *description; } cases[] = { /* --- No credentials at all --- */ { "/api/settings", "", "no auth – /api/settings" }, { "/api/wifi", "", "no auth – /api/wifi" }, { "/api/files", "", "no auth – /api/files" }, /* --- Malformed / garbage tokens --- */ { "/api/settings", "Authorization: Bearer INVALID_TOKEN", "bad bearer – /api/settings" }, { "/api/wifi", "Authorization: Bearer !!GARBAGE!!", "garbage bearer – /api/wifi" }, { "/api/files", "Authorization: Bearer null", "null bearer – /api/files" }, /* --- Expired / obviously fake JWT-like tokens --- */ { "/api/settings", "Authorization: Bearer eyJhbGciOiJub25lIn0.eyJleHAiOjF9.", "expired JWT (alg=none) – /api/settings" }, { "/api/wifi", "Authorization: Bearer eyJhbGciOiJIUzI1NiJ9.eyJzdWIiOiJhZG1pbiIsImV4cCI6MX0.FAKESIG", "expired JWT – /api/wifi" }, /* --- SQL-injection style payloads in token --- */ { "/api/settings", "Authorization: Bearer ' OR '1'='1", "SQLi in token – /api/settings" }, { "/api/wifi", "Authorization: Bearer admin'--", "SQLi comment in token – /api/wifi" }, /* --- Basic auth with wrong credentials --- */ { "/api/settings", "Authorization: Basic YWRtaW46d3Jvbmdfcw==", /* admin:wrong_s */ "wrong basic auth – /api/settings" }, { "/api/wifi", "Authorization: Basic aGFja2VyOmhhY2tlcg==", /* hacker:hacker */ "attacker basic auth – /api/wifi" }, /* --- Cookie-based attack payloads (no valid session) --- */ { "/api/settings", "Cookie: session=", "empty session cookie – /api/settings" }, { "/api/wifi", "Cookie: session=FORGED_SESSION_ID", "forged session – /api/wifi" }, /* --- X-API-Key style attacks --- */ { "/api/settings", "X-API-Key: 00000000-0000-0000-0000-000000000000", "null UUID api key – /api/settings" }, { "/api/wifi", "X-API-Key: ' OR 1=1--", "SQLi api key – /api/wifi" }, }; int num_cases = (int)(sizeof(cases) / sizeof(cases[0])); int verified_cases = 0; for (int i = 0; i < num_cases; i++) { char request[BUFFER_SIZE]; build_request(request, sizeof(request), cases[i].path, cases[i].auth_header); int status = send_http_request(request); if (status == -2) { /* Server not reachable – mark test as skipped, not failed */ fprintf(stderr, "[SKIP] Server not reachable at %s:%d – " "cannot verify: %s\n", SERVER_HOST, SERVER_PORT, cases[i].description); continue; } if (status == -1) { /* Communication error – treat as inconclusive, not a pass */ fprintf(stderr, "[WARN] Communication error for case: %s\n", cases[i].description); continue; } verified_cases++; fprintf(stderr, "[INFO] %s => HTTP %d\n", cases[i].description, status); /* * SECURITY INVARIANT: * A 200 (or any 2xx) response to an unauthenticated request * means the endpoint is completely unprotected – this is the * vulnerability described in CWE-287. * * Acceptable responses: 401 Unauthorized, 403 Forbidden, * 301/302 redirect to login, 404 (hidden), * 405 Method Not Allowed. * Unacceptable: 2xx (data returned without auth). */ ck_assert_msg( status != 200 && !(status >= 200 && status < 300), "SECURITY VIOLATION (CWE-287): endpoint '%s' returned HTTP %d " "for unauthenticated request [case: %s]. " "Expected 401 or 403.", cases[i].path, status, cases[i].description ); } ck_assert_msg(verified_cases > 0, "SECURITY TEST INCONCLUSIVE: zero cases were verified " "(server unreachable or all communications failed). " "Run this test with the device accessible at %s:%d.", SERVER_HOST, SERVER_PORT); } END_TEST /* ------------------------------------------------------------------ */ /* Additional focused test: POST to write endpoints without auth */ /* ------------------------------------------------------------------ */ START_TEST(test_write_endpoints_reject_unauthenticated_post) { /* * Invariant: Write/mutating endpoints must not accept POST requests * without valid authentication. */ struct { const char *path; const char *body; const char *description; } post_cases[] = { { "/api/wifi", "{\"ssid\":\"evil\",\"password\":\"evil123\"}", "POST /api/wifi no auth" }, { "/api/wifi/delete", "{\"index\":0}", "POST /api/wifi/delete no auth" }, { "/api/settings", "{\"admin_password\":\"hacked\"}", "POST /api/settings no auth" }, { "/api/files", "filename=../../etc/passwd&content=hacked", "POST /api/files path traversal no auth" }, }; int num_cases = (int)(sizeof(post_cases) / sizeof(post_cases[0])); int verified_cases = 0; for (int i = 0; i < num_cases; i++) { char request[BUFFER_SIZE]; snprintf(request, sizeof(request), "POST %s HTTP/1.1\r\n" "Host: %s\r\n" "Content-Type: application/json\r\n" "Content-Length: %zu\r\n" "Connection: close\r\n" "\r\n" "%s", post_cases[i].path, SERVER_HOST, strlen(post_cases[i].body), post_cases[i].body); int status = send_http_request(request); if (status == -2) { fprintf(stderr, "[SKIP] Server not reachable – cannot verify: %s\n", post_cases[i].description); continue; } if (status == -1) { fprintf(stderr, "[WARN] Communication error for case: %s\n", post_cases[i].description); continue; } verified_cases++; fprintf(stderr, "[INFO] %s => HTTP %d\n", post_cases[i].description, status); ck_assert_msg( !(status >= 200 && status < 300), "SECURITY VIOLATION (CWE-287): write endpoint '%s' returned " "HTTP %d for unauthenticated POST [case: %s]. " "Expected 401 or 403.", post_cases[i].path, status, post_cases[i].description ); } ck_assert_msg(verified_cases > 0, "SECURITY TEST INCONCLUSIVE: zero cases were verified " "(server unreachable or all communications failed). " "Run this test with the device accessible at %s:%d.", SERVER_HOST, SERVER_PORT); } END_TEST /* ------------------------------------------------------------------ */ Suite *security_suite(void) { Suite *s; TCase *tc_core; s = suite_create("Security_CWE287_AuthenticationBypass"); tc_core = tcase_create("Core"); tcase_set_timeout(tc_core, 30); tcase_add_test(tc_core, test_unauthenticated_requests_rejected); tcase_add_test(tc_core, test_write_endpoints_reject_unauthenticated_post); suite_add_tcase(s, tc_core); return s; } int main(void) { int number_failed; Suite *s; SRunner *sr; s = security_suite(); sr = srunner_create(s); srunner_run_all(sr, CK_NORMAL); number_failed = srunner_ntests_failed(sr); srunner_free(sr); return (number_failed == 0) ? EXIT_SUCCESS : EXIT_FAILURE; }
